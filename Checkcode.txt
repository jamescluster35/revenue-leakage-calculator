function getAdminPassword() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Config');
  if (!sheet) return "SHEET_NOT_FOUND";
  return String(sheet.getRange('A1').getValue());
}

function doGet(e) { return handleRequest(e); }
function doPost(e) { return handleRequest(e); }

function handleRequest(e) {
  const event = e || {};
  const data = (event.postData && event.postData.contents) ? JSON.parse(event.postData.contents) : {};
  let action = data.action || (event.parameter ? event.parameter.action : null);
  const password = data.password || "";

  // Detect a webhook payload when no explicit action is provided
  if (!action && data && data.data && data.data.id && data.data.attributes) {
    action = 'webhookPayment';
  }

  // Password Protection Logic
  const protectedActions = ['getAll', 'moveLead', 'deleteLead', 'archiveLead', 'promoteClient'];
  if (protectedActions.includes(action) && password !== getAdminPassword()) {
    return ContentService.createTextOutput(JSON.stringify({ error: 'Unauthorized: Access Denied' }))
      .setMimeType(ContentService.MimeType.JSON);
  }

  // 3. Routing Logic
  try {
    let result;
    switch (action) {
      case 'getAll':               result = getAll(); break;
      case 'addLead':              result = addLead(data.lead); break;
      case 'updateLead':           result = updateLead(data.id, data.changes, data.tab || 'Leads'); break;
      case 'moveLead':             result = moveLead(data.id, data.fromTab, data.toTab, data.changes || {}); break;
      case 'deleteLead':           result = moveLead(data.id, data.fromTab, 'Deleted', {}); break;
      case 'archiveLead':          result = moveLead(data.id, 'Leads', 'Archived', {}); break;
      case 'promoteClient':        result = moveLead(data.id, 'Leads', 'Clients', {}); break;
      case 'restoreLead':          result = moveLead(data.id, data.fromTab, 'Leads', { status: 'New' }); break;
      case 'saveCalculatorLead':    result = saveCalculatorLead(data.lead); break;
      case 'verifyPayment':         result = verifyPayment(data.orderId, data.email); break;
      case 'webhookPayment':        result = webhookPayment(data); break;
      case 'getCalculatorLeads':    result = getCalculatorLeads(); break;
      case 'updateCalculatorLead':  result = updateCalculatorLead(data.id || data.email, data.changes || {}); break;
      case 'markPaymentPaid':       result = markPaymentPaid(data.id || data.email, data.changes || {}); break;
      case 'deleteCalculatorLead':  result = deleteCalculatorLead(data.id || data.email); break;
      case 'generateAndSendReport': result = generateAndSendReport(data.lead, data.email, data.note, data.subject, data.htmlBody); break;
      case 'sendFollowUpEmail1':       result = sendFollowUpEmailDirect(data.lead, data.email); break;
      case 'sendFollowUpEmails':        result = sendFollowUpEmails(data.lead, data.email); break;
      case 'sendPaymentRequestEmail':   result = sendPaymentRequestEmail(data.lead, data.email, data.wiseLink); break;
      default: result = { error: 'Unknown action: ' + action };
    }
    return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ error: err.message, stack: err.stack }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
// ── Get all tabs data ──────────────────────────────────
function getAll() {
  return {
    leads:    getTabData('Leads'),
    archived: getTabData('Archived'),
    deleted:  getTabData('Deleted'),
    clients:  getTabData('Clients'),
  }
}

function getTabData(tabName) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(tabName)
  if (!sheet) return []
  const rows = sheet.getDataRange().getValues()
  if (rows.length <= 1) return []
  const headers = rows[0]
  return rows.slice(1).map(row => {
    const lead = {}
    headers.forEach((h, i) => {
      if (h === 'outreachLog' || h === 'contacts') {
        try { lead[h] = JSON.parse(row[i] || '[]') }
        catch { lead[h] = [] }
      } else if (h === 'pitchSent') {
        lead[h] = row[i] === true || row[i] === 'TRUE' || row[i] === 'true'
      } else {
        lead[h] = row[i] === undefined ? '' : row[i]
      }
    })
    return lead
  }).filter(l => l.id && l.id !== '')
}

// ── Add lead to Leads tab ──────────────────────────────
function addLead(lead) {
  const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Leads')
  const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]
  const row     = headers.map(h => {
    if (h === 'outreachLog' || h === 'contacts') return JSON.stringify(lead[h] || [])
    if (h === 'pitchSent') return lead[h] ? 'TRUE' : 'FALSE'
    return lead[h] !== undefined ? lead[h] : ''
  })
  sheet.appendRow(row)
  return { success: true }
}

// ── Update lead in any tab ─────────────────────────────
function updateLead(id, changes, tabName) {
  const sheet   = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(tabName)
  if (!sheet) return { error: 'Tab not found: ' + tabName }
  const data    = sheet.getDataRange().getValues()
  const headers = data[0]
  const idCol   = headers.indexOf('id')

  for (let i = 1; i < data.length; i++) {
    if (String(data[i][idCol]) === String(id)) {
      Object.keys(changes).forEach(key => {
        const col = headers.indexOf(key)
        if (col !== -1) {
          let val = changes[key]
          if (key === 'outreachLog' || key === 'contacts') val = JSON.stringify(val)
          if (key === 'pitchSent') val = val ? 'TRUE' : 'FALSE'
          sheet.getRange(i + 1, col + 1).setValue(val)
        }
      })
      return { success: true }
    }
  }
  return { error: 'Lead not found in ' + tabName }
}

// ── Move lead between tabs ─────────────────────────────
function moveLead(id, fromTab, toTab, changes) {
  const ss       = SpreadsheetApp.getActiveSpreadsheet()
  const fromSheet = ss.getSheetByName(fromTab)
  const toSheet   = ss.getSheetByName(toTab)
  if (!fromSheet) return { error: 'Source tab not found: ' + fromTab }
  if (!toSheet)   return { error: 'Destination tab not found: ' + toTab }

  const fromData    = fromSheet.getDataRange().getValues()
  const fromHeaders = fromData[0]
  const toHeaders   = toSheet.getRange(1, 1, 1, toSheet.getLastColumn()).getValues()[0]
  const idCol       = fromHeaders.indexOf('id')

  for (let i = 1; i < fromData.length; i++) {
    if (String(fromData[i][idCol]) === String(id)) {
      // Build lead object from source row
      const lead = {}
      fromHeaders.forEach((h, j) => {
        if (h === 'outreachLog' || h === 'contacts') {
          try { lead[h] = JSON.parse(fromData[i][j] || '[]') }
          catch { lead[h] = [] }
        } else {
          lead[h] = fromData[i][j]
        }
      })

      // Apply any changes
      Object.assign(lead, changes)

      // Build row for destination tab
      const newRow = toHeaders.map(h => {
        if (h === 'outreachLog' || h === 'contacts') return JSON.stringify(lead[h] || [])
        if (h === 'pitchSent') return lead[h] ? 'TRUE' : 'FALSE'
        return lead[h] !== undefined ? lead[h] : ''
      })

      // Add to destination, delete from source
      toSheet.appendRow(newRow)
      fromSheet.deleteRow(i + 1)
      return { success: true }
    }
  }
  return { error: 'Lead not found in ' + fromTab }
}

function deleteCalculatorLead(identifier) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName('Calculator Leads')
  if (!sheet) return { error: 'Sheet not found' }
  const data = sheet.getDataRange().getValues()
  const headers = data[0]
  const idCol = headers.indexOf('id')
  const emailCol = headers.indexOf('email')
  const lookupValue = String(identifier || '').trim()
  const isEmail = lookupValue.includes('@')
  if (!lookupValue) return { error: 'No identifier provided' }

  for (let i = 1; i < data.length; i++) {
    const rowId = idCol !== -1 ? String(data[i][idCol]).trim() : ''
    const rowEmail = emailCol !== -1 ? String(data[i][emailCol]).trim().toLowerCase() : ''
    const matched = (isEmail && rowEmail === lookupValue.toLowerCase()) || (!isEmail && rowId === lookupValue)
    if (matched) {
      sheet.deleteRow(i + 1)
      return { success: true }
    }
  }
  return { error: 'Lead not found' }
}

function createPdfAttachment(html, filename) {
  const blob = Utilities.newBlob(html, 'text/html', filename || 'report.html')
  const pdfBlob = blob.getAs('application/pdf')
  pdfBlob.setName((filename || 'report').replace(/\.html?$/i, '') + '.pdf')
  return pdfBlob
}
